122static bool topicMatches(const Submission& sub, const Reviewer& rev, int genMode, int& matchDomain) {
123 if (genMode <= 1) {
124 if (sub.primaryTopic == rev.primaryExpertise) { matchDomain = sub.primaryTopic; return true; }
125 return false;
126 }
127 // mode 2: primary + secondary submission domain vs primary reviewer
128 if (genMode == 2) {
129 if (sub.primaryTopic == rev.primaryExpertise) { matchDomain = sub.primaryTopic; return true; }
130 if (sub.secondaryTopic > 0 && sub.secondaryTopic == rev.primaryExpertise) { matchDomain = sub.secondaryTopic; return true; }
131 return false;
132 }
133 // mode 3: all combinations
134 if (sub.primaryTopic == rev.primaryExpertise) { matchDomain = sub.primaryTopic; return true; }
135 if (rev.secondaryExpertise > 0 && sub.primaryTopic == rev.secondaryExpertise) { matchDomain = sub.primaryTopic; return true; }
136 if (sub.secondaryTopic > 0 && sub.secondaryTopic == rev.primaryExpertise) { matchDomain = sub.secondaryTopic; return true; }
137 if (sub.secondaryTopic > 0 && rev.secondaryExpertise > 0 && sub.secondaryTopic == rev.secondaryExpertise) { matchDomain = sub.secondaryTopic; return true; }
138 return false;
139}
