172static int batchMode(const string& inputFile, const string& outputFile) {
173 ParseResult data = parseInputFile(inputFile);
174 if (!data.success) { cerr << "[ERROR] Failed to parse: " << inputFile << endl; return 1; }
175
176 ofstream out(outputFile);
177 if (!out.is_open()) { cerr << "[ERROR] Cannot write: " << outputFile << endl; return 1; }
178
179 int gm = data.control.generateAssignments; if (gm == 0) gm = 1;
180 auto res = solveAssignment(data.submissions, data.reviewers, data.parameters, gm);
181
182 bool doRisk = (data.control.riskAnalysis >= 1);
183 RiskAnalysisResult riskResult;
184 if (doRisk) riskResult = runRiskAnalysis1(data.submissions, data.reviewers, data.parameters, gm);
185
186 writeResults(out, res, riskResult, doRisk);
187 out.close();
188 cout << "[OK] Output: " << outputFile << endl;
189 return 0;
190}
