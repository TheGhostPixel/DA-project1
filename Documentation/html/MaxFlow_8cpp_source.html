48bool bfs(Graph<int>& g, int source, int sink) {
49 for (auto v : g.getVertexSet())

70 }

84static double findMinResidualAlongPath(Vertex<int>* s, Vertex<int>* t) {
85 double f = 1e18;
86 for (auto v = t; v != s; ) {
87 auto e = v->getPath();
