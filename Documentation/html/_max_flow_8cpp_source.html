20// BFS to find an augmenting path from source to sink
21// We check outgoing edges (forward) and incoming edges (backward)
22// Same idea as TP3 findAugmentingPath
23bool bfs(Graph<int>& g, int source, int sink) {
24 for (auto v : g.getVertexSet())
25 v->setVisited(false);
26
27 auto s = g.findVertex(source);
28 auto t = g.findVertex(sink);
29 s->setVisited(true);
30
31 std::queue<Vertex<int>*> q;
32 q.push(s);
33
34 while (!q.empty() && !t->isVisited()) {
35 auto v = q.front();
36 q.pop();
37
38 // forward edges: residual = weight - flow
39 for (auto e : v->getAdj())
40 testAndVisit(q, e, e->getDest(), e->getWeight() - e->getFlow());
41
42 // backward edges: residual = flow
43 for (auto e : v->getIncoming())
44 testAndVisit(q, e, e->getOrig(), e->getFlow());
45 }
46
47 return t->isVisited();
48}

50// Walk from sink to source along the path, find the smallest residual
51// Same as TP3 findMinResidualAlongPath
52static double findMinResidualAlongPath(Vertex<int>* s, Vertex<int>* t) {
53 double f = 1e18;
54 for (auto v = t; v != s; ) {
55 auto e = v->getPath();
56 if (e->getDest() == v) {
57 f = std::min(f, e->getWeight() - e->getFlow());
58 v = e->getOrig();
59 } else {
60 f = std::min(f, e->getFlow());
61 v = e->getDest();
62 }
63 }
64 return f;
65}

67// Push flow along the path
68// Same as TP3 augmentFlowAlongPath
69static void augmentFlowAlongPath(Vertex<int>* s, Vertex<int>* t, double f) {
70 for (auto v = t; v != s; ) {
71 auto e = v->getPath();
72 double flow = e->getFlow();
73 if (e->getDest() == v) {
74 e->setFlow(flow + f);
75 v = e->getOrig();
76 } else {
77 e->setFlow(flow - f);
78 v = e->getDest();
79 }
80 }

83// Edmonds-Karp: keep finding augmenting paths with BFS and pushing flow
84// Same structure as TP3 edmondsKarp
85// O(V * E^2)
86double edmondsKarp(Graph<int>& g, int source, int sink) {
87 // set all flows to 0
88 for (auto v : g.getVertexSet())
89 for (auto e : v->getAdj())
90 e->setFlow(0);
91
92 auto s = g.findVertex(source);
93 auto t = g.findVertex(sink);
94 double maxFlow = 0;
95
96 while (bfs(g, source, sink)) {
97 double f = findMinResidualAlongPath(s, t);
98 augmentFlowAlongPath(s, t, f);
99 maxFlow += f;
100 }
101
102 return maxFlow;
103}
